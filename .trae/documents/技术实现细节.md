## 核心功能实现

### 1. 用户认证中间件 (server/middleware/auth.js)
```javascript
const jwt = require('jsonwebtoken');
const JWT_SECRET = process.env.JWT_SECRET || 'your-secret-key';

const authenticateToken = (req, res, next) => {
  const authHeader = req.headers['authorization'];
  const token = authHeader && authHeader.split(' ')[1];

  if (!token) {
    return res.status(401).json({ error: '访问令牌缺失' });
  }

  jwt.verify(token, JWT_SECRET, (err, user) => {
    if (err) {
      return res.status(403).json({ error: '令牌无效' });
    }
    req.user = user;
    next();
  });
};

module.exports = { authenticateToken };
```

### 4. React上下文管理 (client/src/context/AuthContext.js)
```javascript
import React, { createContext, useContext, useState, useEffect } from 'react';
import { authAPI } from '../services/api';

const AuthContext = createContext();

export const useAuth = () => {
  const context = useContext(AuthContext);
  if (!context) {
    throw new Error('useAuth必须在AuthProvider内使用');
  }
  return context;
};

export const AuthProvider = ({ children }) => {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const token = localStorage.getItem('token');
    if (token) {
      // 验证token并获取用户信息
      verifyToken();
    } else {
      setLoading(false);
    }
  }, []);

  const verifyToken = async () => {
    try {
      // 这里需要后端提供一个验证token的接口
      const response = await authAPI.verify();
      setUser(response.data.user);
    } catch (error) {
      localStorage.removeItem('token');
    } finally {
      setLoading(false);
    }
  };

  const login = async (credentials) => {
    const response = await authAPI.login(credentials);
    const { token, user } = response.data;
    
    localStorage.setItem('token', token);
    setUser(user);
    
    return response;
  };

  const logout = () => {
    localStorage.removeItem('token');
    setUser(null);
  };

  const register = async (userData) => {
    return await authAPI.register(userData);
  };

  const value = {
    user,
    login,
    logout,
    register,
    loading,
  };

  return (
    <AuthContext.Provider value={value}>
      {children}
    </AuthContext.Provider>
  );
};
```

### 统一错误处理中间件
```javascript
// server/middleware/errorHandler.js
const errorHandler = (err, req, res, next) => {
  console.error('Error:', err);

  // 数据库错误
  if (err.code === 'SQLITE_CONSTRAINT') {
    return res.status(400).json({
      error: '数据约束违反',
      message: '请检查输入数据是否重复或无效'
    });
  }

  // JWT错误
  if (err.name === 'JsonWebTokenError') {
    return res.status(401).json({
      error: '认证失败',
      message: '令牌无效'
    });
  }

  // 文件上传错误
  if (err.code === 'MULTER_FILE_TOO_LARGE') {
    return res.status(400).json({
      error: '文件太大',
      message: '请选择小于10MB的图片或100MB的视频'
    });
  }

  // 默认服务器错误
  res.status(500).json({
    error: '服务器内部错误',
    message: process.env.NODE_ENV === 'development' ? err.message : '请稍后重试'
  });
};

module.exports = errorHandler;
```

## 媒体文件上传实现

### 1. 服务器端文件上传处理 (server/routes/upload.js)

```javascript
const express = require('express');
const multer = require('multer');
const path = require('path');
const fs = require('fs');
const ffmpeg = require('fluent-ffmpeg');
const ffmpegStatic = require('ffmpeg-static');
const { authenticateToken } = require('../middleware/auth');
const { db } = require('../database');

const router = express.Router();

// 设置ffmpeg路径
ffmpeg.setFfmpegPath(ffmpegStatic);

// 配置存储
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    const uploadDir = file.mimetype.startsWith('video/') ? 'uploads/videos' : 'uploads/photos';
    if (!fs.existsSync(uploadDir)) {
      fs.mkdirSync(uploadDir, { recursive: true });
    }
    cb(null, uploadDir);
  },
  filename: (req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, file.fieldname + '-' + uniqueSuffix + path.extname(file.originalname));
  }
});

// 文件类型过滤
const fileFilter = (req, file, cb) => {
  if (file.mimetype.startsWith('image/') || file.mimetype.startsWith('video/')) {
    cb(null, true);
  } else {
    cb(new Error('只支持图片和视频文件'), false);
  }
};

// 配置multer - 根据用户等级设置不同限制
const upload = multer({
  storage: storage,
  fileFilter: fileFilter,
  limits: {
    fileSize: (req, res, next) => {
      // 这里可以根据用户VIP状态动态设置限制
      const isVip = req.user?.isVip || false;
      const photoLimit = isVip ? 20 * 1024 * 1024 : 10 * 1024 * 1024; // VIP: 20MB, 普通: 10MB
      const videoLimit = isVip ? 500 * 1024 * 1024 : 100 * 1024 * 1024; // VIP: 500MB, 普通: 100MB
      
      return req.file?.mimetype.startsWith('video/') ? videoLimit : photoLimit;
    }
  }
});

// 生成视频缩略图
const generateThumbnail = (videoPath, outputPath) => {
  return new Promise((resolve, reject) => {
    ffmpeg(videoPath)
      .screenshots({
        timestamps: ['50%'],
        filename: path.basename(outputPath),
        folder: path.dirname(outputPath),
        size: '320x240'
      })
      .on('end', () => resolve(outputPath))
      .on('error', reject);
  });
};

// 获取视频时长
const getVideoDuration = (videoPath) => {
  return new Promise((resolve, reject) => {
    ffmpeg.ffprobe(videoPath, (err, metadata) => {
      if (err) reject(err);
      else resolve(Math.floor(metadata.format.duration));
    });
  });
};

// 检查用户存储空间
const checkStorageLimit = async (userId, fileSize) => {
  const userQuery = `SELECT is_vip, used_storage FROM users WHERE id = ?`;
  const user = await new Promise((resolve, reject) => {
    db.get(userQuery, [userId], (err, row) => {
      if (err) reject(err);
      else resolve(row);
    });
  });

  const maxStorage = user.is_vip ? 10 * 1024 * 1024 * 1024 : 1024 * 1024 * 1024; // VIP: 10GB, 普通: 1GB
  const usedStorage = user.used_storage || 0;

  if (usedStorage + fileSize > maxStorage) {
    throw new Error(`存储空间不足。${user.is_vip ? 'VIP用户' : '免费用户'}最大存储: ${maxStorage / (1024 * 1024 * 1024)}GB`);
  }

  return true;
};

// 更新用户已使用存储空间
const updateUsedStorage = async (userId, fileSize) => {
  const updateQuery = `UPDATE users SET used_storage = COALESCE(used_storage, 0) + ? WHERE id = ?`;
  return new Promise((resolve, reject) => {
    db.run(updateQuery, [fileSize, userId], (err) => {
      if (err) reject(err);
      else resolve();
    });
  });
};

// 照片上传
router.post('/photo', authenticateToken, upload.single('photo'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: '请选择图片文件' });
    }

    // 检查存储限制
    await checkStorageLimit(req.user.id, req.file.size);

    const mediaData = {
      couple_id: req.body.couple_id,
      record_id: req.body.record_id || null,
      file_url: req.file.path,
      file_type: 'photo',
      caption: req.body.caption || null,
      file_size: req.file.size,
      mime_type: req.file.mimetype
    };

    const query = `INSERT INTO media (couple_id, record_id, file_url, file_type, caption, file_size, mime_type) 
                   VALUES (?, ?, ?, ?, ?, ?, ?)`;
    
    const mediaId = await new Promise((resolve, reject) => {
      db.run(query, Object.values(mediaData), function(err) {
        if (err) reject(err);
        else resolve(this.lastID);
      });
    });

    // 更新存储使用量
    await updateUsedStorage(req.user.id, req.file.size);

    res.json({ 
      success: true, 
      mediaId,
      fileUrl: req.file.path,
      fileType: 'photo'
    });

  } catch (error) {
    // 删除已上传的文件
    if (req.file && fs.existsSync(req.file.path)) {
      fs.unlinkSync(req.file.path);
    }
    res.status(400).json({ error: error.message });
  }
});

// 视频上传
router.post('/video', authenticateToken, upload.single('video'), async (req, res) => {
  try {
    if (!req.file) {
      return res.status(400).json({ error: '请选择视频文件' });
    }

    // 检查存储限制
    await checkStorageLimit(req.user.id, req.file.size);

    // 获取视频时长
    const duration = await getVideoDuration(req.file.path);
    
    // 生成缩略图
    const thumbnailPath = req.file.path.replace(path.extname(req.file.path), '_thumb.jpg');
    await generateThumbnail(req.file.path, thumbnailPath);

    const mediaData = {
      couple_id: req.body.couple_id,
      record_id: req.body.record_id || null,
      file_url: req.file.path,
      file_type: 'video',
      caption: req.body.caption || null,
      file_size: req.file.size,
      mime_type: req.file.mimetype,
      duration: duration,
      thumbnail_url: thumbnailPath
    };

    const query = `INSERT INTO media (couple_id, record_id, file_url, file_type, caption, file_size, mime_type, duration, thumbnail_url) 
                   VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)`;
    
    const mediaId = await new Promise((resolve, reject) => {
      db.run(query, Object.values(mediaData), function(err) {
        if (err) reject(err);
        else resolve(this.lastID);
      });
    });

    // 更新存储使用量（包括缩略图）
    const thumbnailSize = fs.statSync(thumbnailPath).size;
    await updateUsedStorage(req.user.id, req.file.size + thumbnailSize);

    res.json({ 
      success: true, 
      mediaId,
      fileUrl: req.file.path,
      thumbnailUrl: thumbnailPath,
      fileType: 'video',
      duration: duration
    });

  } catch (error) {
    // 删除已上传的文件
    if (req.file && fs.existsSync(req.file.path)) {
      fs.unlinkSync(req.file.path);
    }
    res.status(400).json({ error: error.message });
  }
});

// 获取媒体文件列表
router.get('/media/:recordId?', authenticateToken, (req, res) => {
  const { recordId } = req.params;
  const query = recordId 
    ? `SELECT * FROM media WHERE record_id = ? ORDER BY created_at DESC`
    : `SELECT * FROM media WHERE couple_id = ? ORDER BY created_at DESC`;
  
  const params = recordId ? [recordId] : [req.body.couple_id];

  db.all(query, params, (err, rows) => {
    if (err) {
      res.status(500).json({ error: '获取媒体文件失败' });
    } else {
      res.json(rows);
    }
  });
});

// 删除媒体文件
router.delete('/media/:mediaId', authenticateToken, async (req, res) => {
  try {
    const { mediaId } = req.params;
    
    // 获取文件信息
    const media = await new Promise((resolve, reject) => {
      db.get('SELECT * FROM media WHERE id = ?', [mediaId], (err, row) => {
        if (err) reject(err);
        else resolve(row);
      });
    });

    if (!media) {
      return res.status(404).json({ error: '媒体文件不存在' });
    }

    // 删除物理文件
    if (fs.existsSync(media.file_url)) {
      fs.unlinkSync(media.file_url);
    }
    if (media.thumbnail_url && fs.existsSync(media.thumbnail_url)) {
      fs.unlinkSync(media.thumbnail_url);
    }

    // 删除数据库记录
    await new Promise((resolve, reject) => {
      db.run('DELETE FROM media WHERE id = ?', [mediaId], (err) => {
        if (err) reject(err);
        else resolve();
      });
    });

    // 更新存储使用量
    const fileSize = media.file_size + (media.thumbnail_url ? fs.statSync(media.thumbnail_url).size : 0);
    await updateUsedStorage(req.user.id, -fileSize);

    res.json({ success: true });

  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

module.exports = router;
```

### 2. 更新用户表结构


### 3. 存储空间管理中间件 (server/middleware/storage.js)

```javascript
const { db } = require('../database');

// 存储空间限制常量
const STORAGE_LIMITS = {
  FREE: 1 * 1024 * 1024 * 1024,      // 免费用户: 1GB
  VIP: 10 * 1024 * 1024 * 1024,      // VIP用户: 10GB
};

const FILE_SIZE_LIMITS = {
  PHOTO: {
    FREE: 10 * 1024 * 1024,          // 免费用户照片: 10MB
    VIP: 20 * 1024 * 1024,           // VIP用户照片: 20MB
  },
  VIDEO: {
    FREE: 100 * 1024 * 1024,         // 免费用户视频: 100MB
    VIP: 500 * 1024 * 1024,          // VIP用户视频: 500MB
  }
};

// 检查用户存储空间
const checkStorageQuota = async (req, res, next) => {
  try {
    const userId = req.user.id;
    const user = await new Promise((resolve, reject) => {
      db.get('SELECT is_vip, used_storage, vip_expires_at FROM users WHERE id = ?', [userId], (err, row) => {
        if (err) reject(err);
        else resolve(row);
      });
    });

    const isVipActive = user.is_vip && (!user.vip_expires_at || new Date(user.vip_expires_at) > new Date());
    const maxStorage = isVipActive ? STORAGE_LIMITS.VIP : STORAGE_LIMITS.FREE;
    const usedStorage = user.used_storage || 0;

    req.user.storageInfo = {
      isVip: isVipActive,
      maxStorage,
      usedStorage,
      availableStorage: maxStorage - usedStorage
    };

    next();
  } catch (error) {
    res.status(500).json({ error: '存储空间检查失败' });
  }
};

// 获取用户存储信息
const getStorageInfo = (req, res) => {
  const { storageInfo } = req.user;
  res.json({
    isVip: storageInfo.isVip,
    maxStorage: storageInfo.maxStorage,
    usedStorage: storageInfo.usedStorage,
    availableStorage: storageInfo.availableStorage,
    usagePercentage: Math.round((storageInfo.usedStorage / storageInfo.maxStorage) * 100),
    limits: {
      photoSize: storageInfo.isVip ? FILE_SIZE_LIMITS.PHOTO.VIP : FILE_SIZE_LIMITS.PHOTO.FREE,
      videoSize: storageInfo.isVip ? FILE_SIZE_LIMITS.VIDEO.VIP : FILE_SIZE_LIMITS.VIDEO.FREE,
    }
  });
};

module.exports = { checkStorageQuota, getStorageInfo, STORAGE_LIMITS, FILE_SIZE_LIMITS };
```

### 4. 前端媒体上传组件 (client/src/components/MediaUpload.jsx)

```javascript
import React, { useState, useRef } from 'react';
import { mediaAPI } from '../services/api';

const MediaUpload = ({ recordId, coupleId, onUploadSuccess }) => {
  const [uploading, setUploading] = useState(false);
  const [uploadProgress, setUploadProgress] = useState(0);
  const [dragOver, setDragOver] = useState(false);
  const fileInputRef = useRef(null);

  // 文件验证
  const validateFile = (file) => {
    const maxSizes = {
      photo: 20 * 1024 * 1024, // 20MB (VIP限制，实际限制由后端根据用户状态确定)
      video: 500 * 1024 * 1024, // 500MB
    };

    const supportedTypes = {
      photo: ['image/jpeg', 'image/png', 'image/gif', 'image/webp'],
      video: ['video/mp4', 'video/webm', 'video/quicktime', 'video/x-msvideo'],
    };

    const fileType = file.type.startsWith('image/') ? 'photo' : 'video';
    
    if (!supportedTypes[fileType].includes(file.type)) {
      throw new Error(`不支持的文件类型: ${file.type}`);
    }

    if (file.size > maxSizes[fileType]) {
      throw new Error(`文件太大，${fileType === 'photo' ? '图片' : '视频'}最大${maxSizes[fileType] / (1024 * 1024)}MB`);
    }

    return fileType;
  };

  // 处理文件上传
  const handleFileUpload = async (files) => {
    if (!files || files.length === 0) return;

    setUploading(true);
    setUploadProgress(0);

    try {
      for (let file of files) {
        const fileType = validateFile(file);
        const formData = new FormData();
        formData.append(fileType, file);
        formData.append('record_id', recordId);
        formData.append('couple_id', coupleId);

        // 上传文件
        const uploadAPI = fileType === 'photo' ? mediaAPI.uploadPhoto : mediaAPI.uploadVideo;
        const response = await uploadAPI(formData);

        // 通知父组件上传成功
        if (onUploadSuccess) {
          onUploadSuccess({
            ...response.data,
            file: file,
            fileType: fileType
          });
        }
      }
    } catch (error) {
      console.error('Upload failed:', error);
      alert(error.message || '上传失败，请重试');
    } finally {
      setUploading(false);
      setUploadProgress(0);
    }
  };

  // 拖拽处理
  const handleDragOver = (e) => {
    e.preventDefault();
    setDragOver(true);
  };

  const handleDragLeave = (e) => {
    e.preventDefault();
    setDragOver(false);
  };

  const handleDrop = (e) => {
    e.preventDefault();
    setDragOver(false);
    const files = Array.from(e.dataTransfer.files);
    handleFileUpload(files);
  };

  // 点击上传
  const handleFileSelect = (e) => {
    const files = Array.from(e.target.files);
    handleFileUpload(files);
  };

  const triggerFileSelect = () => {
    fileInputRef.current?.click();
  };

  return (
    <div className="media-upload">
      {/* 上传区域 */}
      <div
        className={`upload-zone ${dragOver ? 'drag-over' : ''} ${uploading ? 'uploading' : ''}`}
        onDragOver={handleDragOver}
        onDragLeave={handleDragLeave}
        onDrop={handleDrop}
        onClick={triggerFileSelect}
      >
        <input
          type="file"
          ref={fileInputRef}
          multiple
          accept="image/*,video/*"
          onChange={handleFileSelect}
          style={{ display: 'none' }}
        />

        {uploading ? (
          <div className="upload-progress">
            <div className="spinner"></div>
            <p>上传中...</p>
            {uploadProgress > 0 && (
              <div className="progress-bar">
                <div 
                  className="progress-fill" 
                  style={{ width: `${uploadProgress}%` }}
                ></div>
              </div>
            )}
          </div>
        ) : (
          <div className="upload-content">
            <div className="upload-icon">
              <svg width="48" height="48" viewBox="0 0 24 24" fill="none" stroke="currentColor">
                <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4" />
                <polyline points="7,10 12,15 17,10" />
                <line x1="12" y1="15" x2="12" y2="3" />
              </svg>
            </div>
            <p className="upload-text">
              点击或拖拽添加照片/视频
            </p>
            <p className="upload-hint">
              支持 JPG, PNG, GIF, WebP, MP4, WebM 格式
            </p>
          </div>
        )}
      </div>

      {/* 上传按钮 */}
      <div className="upload-buttons">
        <button 
          className="btn-photo" 
          onClick={() => {
            fileInputRef.current.accept = 'image/*';
            triggerFileSelect();
          }}
          disabled={uploading}
        >
          <span>📷</span>
          添加照片
        </button>
        <button 
          className="btn-video" 
          onClick={() => {
            fileInputRef.current.accept = 'video/*';
            triggerFileSelect();
          }}
          disabled={uploading}
        >
          <span>🎥</span>
          添加视频
        </button>
      </div>
    </div>
  );
};

export default MediaUpload;
```

### 5. 媒体展示组件 (client/src/components/MediaGallery.jsx)

```javascript
import React, { useState, useEffect } from 'react';
import { mediaAPI } from '../services/api';

const MediaGallery = ({ recordId, coupleId }) => {
  const [media, setMedia] = useState([]);
  const [loading, setLoading] = useState(true);
  const [selectedMedia, setSelectedMedia] = useState(null);

  useEffect(() => {
    loadMedia();
  }, [recordId]);

  const loadMedia = async () => {
    try {
      setLoading(true);
      const response = await mediaAPI.getMedia(recordId);
      setMedia(response.data);
    } catch (error) {
      console.error('Failed to load media:', error);
    } finally {
      setLoading(false);
    }
  };

  const deleteMedia = async (mediaId) => {
    if (!confirm('确定要删除这个文件吗？')) return;

    try {
      await mediaAPI.deleteMedia(mediaId);
      setMedia(media.filter(item => item.id !== mediaId));
    } catch (error) {
      console.error('Failed to delete media:', error);
      alert('删除失败，请重试');
    }
  };

  const formatFileSize = (bytes) => {
    if (bytes < 1024 * 1024) {
      return `${(bytes / 1024).toFixed(1)} KB`;
    }
    return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
  };

  const formatDuration = (seconds) => {
    const mins = Math.floor(seconds / 60);
    const secs = seconds % 60;
    return `${mins}:${secs.toString().padStart(2, '0')}`;
  };

  if (loading) {
    return <div className="media-loading">加载中...</div>;
  }

  return (
    <div className="media-gallery">
      <div className="media-grid">
        {media.map((item) => (
          <div key={item.id} className="media-item">
            {item.file_type === 'photo' ? (
              <img
                src={item.file_url}
                alt={item.caption}
                className="media-thumbnail"
                onClick={() => setSelectedMedia(item)}
              />
            ) : (
              <div className="video-thumbnail" onClick={() => setSelectedMedia(item)}>
                <img 
                  src={item.thumbnail_url} 
                  alt={item.caption}
                  className="media-thumbnail"
                />
                <div className="video-overlay">
                  <div className="play-button">▶</div>
                  <div className="video-duration">
                    {formatDuration(item.duration)}
                  </div>
                </div>
              </div>
            )}
            
            <div className="media-info">
              <div className="media-caption">{item.caption}</div>
              <div className="media-meta">
                <span className="file-size">{formatFileSize(item.file_size)}</span>
                <button 
                  className="delete-btn"
                  onClick={() => deleteMedia(item.id)}
                  title="删除"
                >
                  🗑️
                </button>
              </div>
            </div>
          </div>
        ))}
      </div>

      {/* 媒体预览模态框 */}
      {selectedMedia && (
        <div className="media-modal" onClick={() => setSelectedMedia(null)}>
          <div className="modal-content" onClick={e => e.stopPropagation()}>
            <button 
              className="modal-close"
              onClick={() => setSelectedMedia(null)}
            >
              ✕
            </button>
            
            {selectedMedia.file_type === 'photo' ? (
              <img 
                src={selectedMedia.file_url} 
                alt={selectedMedia.caption}
                className="modal-image"
              />
            ) : (
              <video 
                src={selectedMedia.file_url}
                controls
                className="modal-video"
                autoPlay
              />
            )}
            
            {selectedMedia.caption && (
              <div className="modal-caption">
                {selectedMedia.caption}
              </div>
            )}
          </div>
        </div>
      )}
    </div>
  );
};
```
